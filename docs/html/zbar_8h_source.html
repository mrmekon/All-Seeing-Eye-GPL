00249         ((unsigned long)(a) |                   \
00250          ((unsigned long)(b) << 8) |            \
00251          ((unsigned long)(c) << 16) |           \
00252          ((unsigned long)(d) << 24))

00257 static inline unsigned long zbar_fourcc_parse (const char *format)
00258 {
00259     unsigned long fourcc = 0;
00260     if(format) {
00261         int i;
00262         for(i = 0; i < 4 && format[i]; i++)
00263             fourcc |= ((unsigned long)format[i]) << (i * 8);
00264     }
00265     return(fourcc);
00266 }
